# Planetary Ley Line Mathematics

## Overview
This document provides the mathematical foundation for implementing planetary-scale ley line circles around the Earth, including great circle calculations, intersection algorithms, and energy distribution models.

## Great Circle Mathematics

### Fundamental Great Circle Equations

Great circles represent the shortest path between two points on a sphere's surface. For ley line circles, we implement the complete mathematical framework:

```typescript
interface GreatCircleParams {
  center: SphericalCoordinate;
  radius: number; // Angular radius in radians
  normalVector: Vector3D;
  planeEquation: PlaneEquation;
}

interface LeyLineCircle extends GreatCircleParams {
  id: string;
  energyLevel: number;
  harmonic: number; // 1st, 2nd, 3rd harmonic circles
  intersectionNodes: IntersectionNode[];
  subsidiaryCircles: LeyLineCircle[]; // Smaller circles generated by intersections

class PlanetaryLeyLineMathematics {
  private static readonly EARTH_RADIUS = 6371000; // meters
  
  /**
   * Calculate great circle from two points
   */
  static greatCircleFromPoints(point1: SphericalCoordinate, point2: SphericalCoordinate): GreatCircleParams {
    // Convert to Cartesian coordinates
    const cart1 = this.sphericalToCartesian(point1);
    const cart2 = this.sphericalToCartesian(point2);
    
    // Normal vector to the plane containing both points and Earth's center
    const normal = this.crossProduct(cart1, cart2);
    const normalizedNormal = this.normalize(normal);
    
    // Center point on the sphere (perpendicular to normal)
    const center = this.findGreatCircleCenter(cart1, cart2, normalizedNormal);
    
    // Angular radius (always π/2 for great circles)
    const radius = Math.PI;
    
    return {
      center: this.cartesianToSpherical(center),
      radius,
      normalVector: normalizedNormal,
      planeEquation: this.planeFromNormalAndPoint(normalizedNormal, { x: 0, y: 0, z: 0 })
    };
  }
  
  /**
   * Generate a ley line circle at specific coordinates with given angular radius
   */
  static generateLeyLineCircle(
    center: SphericalCoordinate, 
    angularRadius: number,
    energyLevel: number = 1.0,
    harmonic: number = 1
  ): LeyLineCircle {
    const centerCartesian = this.sphericalToCartesian(center);
    
    // For small circles (non-great circles), we need a different approach
    const circleNormal = centerCartesian; // Circle normal points toward center
    const normalizedNormal = this.normalize(circleNormal);
    
    return {
      id: this.generateLeyLineId(center, angularRadius, harmonic),
      center,
      radius: angularRadius,
      normalVector: normalizedNormal,
      planeEquation: this.planeFromNormalAndPoint(normalizedNormal, centerCartesian),
      energyLevel,
      harmonic,
      intersectionNodes: [],
      subsidiaryCircles: []
    };
  }
  
  /**
   * Calculate points along a ley line circle
   */
  static sampleLeyLineCircle(circle: LeyLineCircle, numSamples: number): SphericalCoordinate[] {
    const points: SphericalCoordinate[] = [];
    const centerCartesian = this.sphericalToCartesian(circle.center);
    
    // Create two orthogonal vectors in the plane perpendicular to the circle normal
    const normal = circle.normalVector;
    const tangent1 = this.findOrthogonalVector(normal);
    const tangent2 = this.crossProduct(normal, tangent1);
    
    // Normalize tangent vectors
    const normalizedTangent1 = this.normalize(tangent1);
    const normalizedTangent2 = this.normalize(tangent2);
    
    for (let i = 0; i < numSamples; i++) {
      const angle = (2 * Math.PI * i) / numSamples;
      
      // Point on the circle in local coordinates
      const localX = Math.cos(angle) * Math.sin(circle.radius);
      const localY = Math.sin(angle) * Math.sin(circle.radius);
      const localZ = Math.cos(circle.radius);
      
      // Transform to global Cartesian coordinates
      const globalPoint = {
        x: centerCartesian.x * localZ + 
           normalizedTangent1.x * localX + 
           normalizedTangent2.x * localY,
        y: centerCartesian.y * localZ + 
           normalizedTangent1.y * localX + 
           normalizedTangent2.y * localY,
        z: centerCartesian.z * localZ + 
           normalizedTangent1.z * localX + 
           normalizedTangent2.z * localY
      };
      
      // Normalize to sphere surface
      const normalized = this.normalize(globalPoint);
      
      // Convert back to spherical coordinates
      points.push(this.cartesianToSpherical(normalized));
    }
    
    return points;
  }
}
```

### Great Circle Intersection Algorithms

```typescript
class GreatCircleIntersection {
  /**
   * Find intersection points between two great circles
   */
  static findIntersections(circle1: LeyLineCircle, circle2: LeyLineCircle): IntersectionNode[] {
    const normal1 = circle1.normalVector;
    const normal2 = circle2.normalVector;
    
    // Intersection line is perpendicular to both normals
    const intersectionDirection = PlanetaryLeyLineMathematics.crossProduct(normal1, normal2);
    
    // Check if circles are parallel (no intersection or identical)
    const magnitude = PlanetaryLeyLineMathematics.magnitude(intersectionDirection);
    if (magnitude < 1e-10) {
      return []; // Parallel or identical circles
    }
    
    const normalizedDirection = PlanetaryLeyLineMathematics.normalize(intersectionDirection);
    
    // Two intersection points (antipodal)
    const intersection1 = normalizedDirection;
    const intersection2 = {
      x: -normalizedDirection.x,
      y: -normalizedDirection.y,
      z: -normalizedDirection.z
    };
    
    const spherical1 = PlanetaryLeyLineMathematics.cartesianToSpherical(intersection1);
    const spherical2 = PlanetaryLeyLineMathematics.cartesianToSpherical(intersection2);
    
    // Calculate energy levels at intersection points
    const energy1 = this.calculateIntersectionEnergy(circle1, circle2, spherical1);
    const energy2 = this.calculateIntersectionEnergy(circle1, circle2, spherical2);
    
    return [
      {
        id: this.generateIntersectionId(circle1.id, circle2.id, 1),
        position: spherical1,
        connectedCircles: [circle1.id, circle2.id],
        energyLevel: energy1,
        nodeType: this.classifyIntersectionNode(energy1),
        harmonicResonance: this.calculateHarmonicResonance(circle1, circle2)
      },
      {
        id: this.generateIntersectionId(circle1.id, circle2.id, 2),
        position: spherical2,
        connectedCircles: [circle1.id, circle2.id],
        energyLevel: energy2,
        nodeType: this.classifyIntersectionNode(energy2),
        harmonicResonance: this.calculateHarmonicResonance(circle1, circle2)
      }
    ];
  }
  
  /**
   * Calculate energy level at intersection point
   */
  private static calculateIntersectionEnergy(
    circle1: LeyLineCircle, 
    circle2: LeyLineCircle, 
    position: SphericalCoordinate
  ): number {
    // Energy amplification at intersections
    const baseEnergy = (circle1.energyLevel + circle2.energyLevel) / 2;
    
    // Harmonic amplification
    const harmonicFactor = this.calculateHarmonicAmplification(circle1.harmonic, circle2.harmonic);
    
    // Geometric amplification based on intersection angle
    const intersectionAngle = this.calculateIntersectionAngle(circle1, circle2);
    const geometricFactor = Math.sin(intersectionAngle); // Maximum at 90°, minimum at 0°/180°
    
    return baseEnergy * harmonicFactor * geometricFactor;
  }
  
  /**
   * Calculate harmonic amplification between different harmonic levels
   */
  private static calculateHarmonicAmplification(harmonic1: number, harmonic2: number): number {
    if (harmonic1 === harmonic2) {
      // Same harmonic - constructive interference
      return 2.0;
    } else if (harmonic1 % harmonic2 === 0 || harmonic2 % harmonic1 === 0) {
      // Harmonic relationship - moderate amplification
      return 1.5;
    } else if ((harmonic1 + harmonic2) % 2 === 0) {
      // Even sum - slight amplification
      return 1.2;
    } else {
      // No harmonic relationship - minimal amplification
      return 1.1;
    }
  }
  
  /**
   * Calculate angle between two great circles at their intersection
   */
  private static calculateIntersectionAngle(circle1: LeyLineCircle, circle2: LeyLineCircle): number {
    const dot = PlanetaryLeyLineMathematics.dotProduct(circle1.normalVector, circle2.normalVector);
    return Math.acos(Math.abs(dot)); // Absolute value to get acute angle
  }
}
```

```typescript
class LeyLineEnergyDistribution {
  /**
   * Calculate energy distribution across the ley line network
   */
  static calculateEnergyDistribution(network: LeyLineNetwork): EnergyDistributionMap {
    const energyMap = new Map<string, number>();
    
    // Initialize with base circle energies
    for (const circle of [...network.primaryCircles, ...network.secondaryCircles, ...network.tertiaryCircles]) {
      energyMap.set(circle.id, circle.energyLevel);
    }
    
    // Apply intersection amplifications
    for (const node of network.intersectionNodes) {
      const amplificationFactor = this.calculateNodeAmplification(node, network);
      
      // Amplify connected circles
      for (const circleId of node.connectedCircles) {
        const currentEnergy = energyMap.get(circleId) || 0;
        energyMap.set(circleId, currentEnergy * amplificationFactor);
      }
    }
    
    // Calculate energy flow between nodes
    const flowMap = this.calculateEnergyFlow(network, energyMap);
    
    return {
      staticEnergy: energyMap,
      energyFlow: flowMap,
      totalNetworkEnergy: this.calculateTotalEnergy(energyMap),
      energyGradient: this.calculateEnergyGradient(network, energyMap)
    };
  }
  
  /**
   * Calculate energy at any point on the sphere
   */
  static getEnergyAtPoint(position: SphericalCoordinate, network: LeyLineNetwork): number {
    let totalEnergy = 0;
    
    // Energy from proximity to ley line circles
    for (const circle of [...network.primaryCircles, ...network.secondaryCircles, ...network.tertiaryCircles]) {
      const distance = this.distanceToCircle(position, circle);
      const influence = this.calculateEnergyInfluence(distance, circle.energyLevel);
      totalEnergy += influence;
    }
    
    // Energy from proximity to intersection nodes
    for (const node of network.intersectionNodes) {
      const distance = PlanetaryLeyLineMathematics.greatCircleDistance(position, node.position);
      const influence = this.calculateNodeInfluence(distance, node.energyLevel);
      totalEnergy += influence;
    }
    
    // Apply harmonic resonance effects
    const harmonicBonus = this.calculateHarmonicResonanceAt(position, network);
    totalEnergy *= (1 + harmonicBonus);
    
    return totalEnergy;
  }
  
  /**
   * Calculate distance from a point to a ley line circle
   */
  private static distanceToCircle(point: SphericalCoordinate, circle: LeyLineCircle): number {
    const pointCartesian = PlanetaryLeyLineMathematics.sphericalToCartesian(point);
    
    // Distance to the plane containing the circle
    const planeDistance = Math.abs(
      circle.planeEquation.a * pointCartesian.x +
      circle.planeEquation.b * pointCartesian.y +
      circle.planeEquation.c * pointCartesian.z +
      circle.planeEquation.d
    );
    
    // Project point onto the plane
    const projectedPoint = this.projectPointOntoPlane(pointCartesian, circle.planeEquation);
    
    // Distance from projected point to circle center
    const centerCartesian = PlanetaryLeyLineMathematics.sphericalToCartesian(circle.center);
    const distanceToCenter = PlanetaryLeyLineMathematics.magnitude({
      x: projectedPoint.x - centerCartesian.x,
      y: projectedPoint.y - centerCartesian.y,
      z: projectedPoint.z - centerCartesian.z
    });
    
    // Angular distance to circle
    const angularDistanceToCenter = Math.acos(Math.min(1, distanceToCenter));
    const distanceToCircle = Math.abs(angularDistanceToCenter - circle.radius);
    
    return distanceToCircle;
  }
  
  /**
   * Calculate energy influence based on distance
   */
  private static calculateEnergyInfluence(distance: number, maxEnergy: number): number {
    // Exponential decay with distance
    const decayConstant = 0.1; // Adjust for influence range
    return maxEnergy * Math.exp(-distance / decayConstant);
  }
  
  /**
   * Calculate harmonic resonance at a specific point
   */
  private static calculateHarmonicResonanceAt(
    position: SphericalCoordinate, 
    network: LeyLineNetwork
  ): number {
    // Find nearby circles of different harmonics
    const nearbyCircles = this.findNearbyCircles(position, network, 0.1); // 0.1 radian threshold
    
    const harmonics = new Set(nearbyCircles.map(circle => circle.harmonic));
    
    if (harmonics.size <= 1) {
      return 0; // No harmonic resonance with single harmonic
    }
    
    // Calculate resonance based on harmonic relationships
    let resonance = 0;
    const harmonicArray = Array.from(harmonics);
    
    for (let i = 0; i < harmonicArray.length; i++) {
      for (let j = i + 1; j < harmonicArray.length; j++) {
        const h1 = harmonicArray[i];
        const h2 = harmonicArray[j];
        
        if (h1 % h2 === 0 || h2 % h1 === 0) {
          resonance += 0.5; // Strong harmonic relationship
        } else if ((h1 + h2) % 2 === 0) {
          resonance += 0.2; // Weak harmonic relationship
        }
      }
    }
    
    return Math.min(resonance, 1.0); // Cap at 100% bonus
  }
}
```

```typescript
class SphereOctree {
  private root: SphereOctreeNode;
  private maxDepth: number;
  
  constructor(maxDepth: number) {
    this.maxDepth = maxDepth;
    this.root = new SphereOctreeNode(
      { lat: 0, lon: 0, alt: 0 }, // Center of Earth
      Math.PI, // Radius covers entire sphere
      0 // Depth 0
    );
  }
  
  insert(circle: LeyLineCircle): void {
    this.root.insert(circle, this.maxDepth);
  }
  
  query(position: SphericalCoordinate, radius: number): LeyLineCircle[] {
    return this.root.query(position, radius);
  }
}

class SphereOctreeNode {
  private center: SphericalCoordinate;
  private radius: number;
  private depth: number;
  private circles: LeyLineCircle[] = [];
  private children: SphereOctreeNode[] = [];
  
  constructor(center: SphericalCoordinate, radius: number, depth: number) {
    this.center = center;
    this.radius = radius;
    this.depth = depth;
  }
  
  insert(circle: LeyLineCircle, maxDepth: number): void {
    if (this.depth >= maxDepth || this.circles.length < 10) {
      this.circles.push(circle);
      return;
    }
    
    if (this.children.length === 0) {
      this.subdivide();
    }
    
    // Insert into appropriate children
    for (const child of this.children) {
      if (this.circleIntersectsNode(circle, child)) {
        child.insert(circle, maxDepth);
      }
    }
  }
  
  query(position: SphericalCoordinate, radius: number): LeyLineCircle[] {
    const results: LeyLineCircle[] = [];
    
    // Check if query intersects this node
    const distance = PlanetaryLeyLineMathematics.greatCircleDistance(position, this.center);
    if (distance > this.radius + radius) {
      return results; // No intersection
    }
    
    // Add circles from this node
    for (const circle of this.circles) {
      const circleDistance = LeyLineEnergyDistribution.distanceToCircle(position, circle);
      if (circleDistance <= radius) {
        results.push(circle);
      }
    }
    
    // Recursively query children
    for (const child of this.children) {
      results.push(...child.query(position, radius));
    }
    
    return results;
  }
  
  private subdivide(): void {
    // Create 8 children for sphere octree
    const halfRadius = this.radius / 2;
    
    // Generate 8 subdivision centers on the sphere
    const subdivisionCenters = this.generateSubdivisionCenters();
    
    for (const center of subdivisionCenters) {
      this.children.push(new SphereOctreeNode(center, halfRadius, this.depth + 1));
    }
  }
  
  private generateSubdivisionCenters(): SphericalCoordinate[] {
    // Implementation for sphere subdivision
    // (Complex spherical geometry - simplified for brevity)
    return [
      { lat: this.center.lat + this.radius/2, lon: this.center.lon + this.radius/2, alt: 0 },

      { lat: this.center.lat - this.radius/2, lon: this.center.lon - this.radius/2, alt: 0 },
      // ... 4 more subdivisions for complete octree
    ];
```

```typescript
  private static buildLeyLineGraph(network: LeyLineNetwork): Map<string, string[]> {
    const graph = new Map<string, string[]>();
    
    // Add intersection nodes as graph vertices
    for (const node of network.intersectionNodes) {
      graph.set(node.id, []);
    }
    
    // Connect nodes that share ley line circles
    for (const node1 of network.intersectionNodes) {
      for (const node2 of network.intersectionNodes) {
        if (node1.id === node2.id) continue;
        
        // Check if nodes share any ley line circles
        const sharedCircles = node1.connectedCircles.filter(
          circleId => node2.connectedCircles.includes(circleId)
        );
        
        if (sharedCircles.length > 0) {
          const connections = graph.get(node1.id) || [];
          connections.push(node2.id);
          graph.set(node1.id, connections);
        }
      }
    }
    
    return graph;
  }
}
```
